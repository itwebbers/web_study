# transform 实现自适应布局 

> 代码演示

```html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <meta http-equiv="X-UA-Compatible" content="ie=edge">
  <title>ECharts</title>
  <script src="https://cdnjs.cloudflare.com/ajax/libs/echarts/4.2.1/echarts-en.common.js"></script>
  <style>
    html, body {
      margin: 0;
      padding: 0;
      width: 100%;
      height: 100%;
      background-color: skyblue;
      overflow: hidden;
    }

    .page-container {
      width: 100%;
      height: 100%;
    }
    /* 缩放的盒子默认样式 */
    .container {
      width: 1920px;
      height: 1080px;
      transform: scale(1, 1);
      background: rgb(0, 0, 0);

    }
    /* 自动缩放的样式 */
    .container-resize {
      position: relative;
      background-color: bisque;
      user-select: none;
      transform-origin: 0 0;
      box-shadow: 0 0 10px 0 rgba(0,0,0,.5);
      transition: all .3s linear;
      overflow: hidden;
    }
    /* 缩放盒子内的继承盒子 */
    .wraper {
      width: 100%;
      height: 100%;
      position: absolute;
      z-index: -1;
      top: 0;
      left: 0;
      bottom: 0;
      right: 0;
    }
  </style>
</head>
<body>
  <!-- 第一层是让div自适应body的宽高 -->
  <div class="page-container">
    <!-- 第二层是要缩放的区域 -->
    <section id="container" class="container">
      <!-- 第三层是让盒子继承缩放后的盒子宽高 -->
      <div class="wraper" style="position: absolute; width: 100%; height: 100%;">
        <!-- 为ECharts准备一个具备大小（宽高）的Dom -->
        <div id="main" style="position: absolute; width: 500px; height: 400px;"></div>
    
        <div id="pie" style="position: absolute; right: 0; width: 500px; height: 400px;"></div>
      </div>
    </section>
  </div>

<script>

  // 加载页面时执行一次
  resizeContainer();

  function resizeContainer() {
    // 获取网页可见区域宽度比
    let scaleWidth = document.body.offsetWidth / 1920;
    // 获取网页可见区域高度比
    let scaleHeight = document.body.offsetHeight / 1080;
    // 获取页面的container元素
    let container = document.querySelector('#container');
    // 根据页面的实际尺寸设置页面的缩放比
    container.style.transform = "scale("+ scaleWidth + ", " + scaleHeight + ")";
    // 给缩放后的盒子添加动画
    container.classList.add('container-resize');
  }

  // 监听浏览器宽度的改变
  window.onresize = function () {
    resizeContainer();
  };
</script>

  <script type="text/javascript">
    // 基于准备好的dom，初始化echarts实例
    var myChart = echarts.init(document.getElementById('main'));

    // 指定图表的配置项和数据
    var option = {
      title: {
        text: 'ECharts 入门示例'
      },
      tooltip: {},
      legend: {
        data: ['销量']
      },
      xAxis: {
        data: ["衬衫", "羊毛衫", "雪纺衫", "裤子", "高跟鞋", "袜子"]
      },
      yAxis: {},
      series: [{
        name: '销量',
        type: 'bar',
        data: [5, 20, 36, 10, 10, 20]
      }]
    };

    // 使用刚指定的配置项和数据显示图表。
    myChart.setOption(option);
  </script>



<script>
  // 绘制图表。
  echarts.init(document.getElementById('pie')).setOption({
    series: {
      type: 'pie',
      data: [
        { name: 'A', value: 1212 },
        { name: 'B', value: 2323 },
        { name: 'C', value: 1919 }
      ]
    }
  });
</script>
</body>
</html>
```
